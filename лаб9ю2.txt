#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <sys/ipc.h>
#include <sys/msg.h>

#define MSG_SIZE 128

// Структура для сообщения
struct message {
    long msg_type;
    char msg_text[MSG_SIZE];
};

int main() {
    key_t key;
    int msgid;
    struct message msg;

    // Генерация ключа для очереди сообщений
    key = ftok("msg_queue", 65);
    if (key == -1) {
        perror("ftok");
        exit(EXIT_FAILURE);
    }

    // Получение идентификатора очереди сообщений
    msgid = msgget(key, 0666);
    if (msgid == -1) {
        perror("msgget");
        exit(EXIT_FAILURE);
    }

    // Отправка сообщения серверу
    msg.msg_type = 1;
    strcpy(msg.msg_text, "Hello from client");
    if (msgsnd(msgid, &msg, sizeof(msg.msg_text), 0) == -1) {
        perror("msgsnd");
        exit(EXIT_FAILURE);
    }

    printf("Message sent to server\n");

    // Получение ответа от сервера
    if (msgrcv(msgid, &msg, sizeof(msg.msg_text), 2, 0) == -1) {
        perror("msgrcv");
        exit(EXIT_FAILURE);
    }

    printf("Server says: %s\n", msg.msg_text);

    // Запись сообщения в файл
    FILE *file = fopen("client_log.txt", "a");
    if (file == NULL) {
        perror("fopen");
        exit(EXIT_FAILURE);
    }
    fprintf(file, "Received: %s\n", msg.msg_text);
    fclose(file);

    return 0;
}
